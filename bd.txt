create database bookriddle;

use bookriddle;

create table funcionario(
matricula varchar(30) not null primary key,
nome varchar(50) not null,
telefone varchar(15) not null,
email varchar(30) not null,
endereco varchar(80) not null,
complemento varchar(30) not null,
cidade varchar(20) not null,
cep varchar(9) not null,
sexo char(1) not null,
data_nascimento date not null,
cargo varchar(30) not null,
cpf varchar(14) not null,
rg varchar(10) not null,
status integer not null);


create table usuario(
login varchar(30) not null,
senha varchar(30) not null,
matricula_funcionario varchar(30),
primary key(login),
foreign key(matricula_funcionario) references funcionario(matricula));

create table livro(
id integer not null auto_increment primary key,
titulo varchar(80) not null,
nome_autor varchar(50) not null,
sobrenome_autor varchar(20) not null,
assunto varchar(30) not null,
edicao varchar(20) not null,
editora varchar(30) not null,
armario varchar(3),
prateleira integer,
ano integer not null,
quantidade integer not null,
paginas integer not null,
url varchar(200));
